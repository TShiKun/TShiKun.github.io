        // 2.解决方案:节流设计一个定时器(短时间内同时大量触发同一函数，按频率执行一次)
        // 适用于搜索框input事件
        function throttle(fn,delay){
            //初始化默认不执行 33-37
            var valid = true;
            return function(){
                if (!valid){
                    return false
                }
            //按照频率执行
                valid = false;
                setTimeout(function(){
                    fn();
                    valid = true;
                },delay);
            }
        }
        // 按照频率2s内只要动了,都会打印
        window.onscroll = throttle(scrollHandle,2000)
        function scrollHandle(){
            var scrollTop = document.documentElement.scrollTop
            console.log("页面滚动了"+scrollTop);
        }
    </script>
